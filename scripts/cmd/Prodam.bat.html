echo.
set /p option=Escolha uma opcao: 

if %option% EQU 1 (
    call :Pings
) else if %option% EQU 2 (
    call :ping1
) else if %option% EQU 3 (
    call :ping2
) else if %option% EQU 4 (
    call :ping3
) else if %option% EQU 5 (
    call :shut
) else if %option% EQU 6 (

) else (

    echo.

)

:ping3
