# -*- coding: utf-8 -*-
#Variáveis: 
# -Case sensitve;
# -Utiliza o = para atribuir;
# -Podem ser do tipo int, float, string, boolean (true or false)

mensagem="Hello Universe"
print(mensagem)

inteiro = 1 #Variavel do tipo inteiro
flutuante= 1.1 #Variavel do tipo flutuante
texto = "Exemplo de string" #Variável do tipo String
booleana = True #Variável do tipo boolean
print(inteiro)
print(flutuante)
print(texto)
print(booleana)
